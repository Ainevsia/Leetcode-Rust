# 746. 使用最小花费爬楼梯 

```cpp
class Solution {
public:
    int minCostClimbingStairs(vector<int>& cost) {
        // dp[i] => 到达i的最小花费
        // dp[i] = min(dp[i-1]+cost[i-1],dp[i-2]+cost[i-2])
        int n=cost.size();
        vector<int>dp(n+1);
        dp[0]=0;dp[1]=0;
        for(int i=2;i<n+1;i++)dp[i] = min(dp[i-1]+cost[i-1],dp[i-2]+cost[i-2]);
        return dp[n];
    }
};
```
